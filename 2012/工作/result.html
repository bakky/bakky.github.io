定义一种返回结果 


public class Result <T> {
     private boolean isSuccess ;
     private String message;
     private T result ;     
     /**
      * 创建一个只包含结果的result
      * @param isSuccess
      */
     public Result( boolean isSuccess){
            this.setSuccess (isSuccess );
     }     
     /**
      * 创建一个不带返回对象的result
      * @param isSuccess
      * @param message
      */
     public Result( boolean isSuccess, String message) {
            this.isSuccess = isSuccess;
            this.message = message;
     }     
     /**
      * 创建一个带返回对象的result
      * @param isSuccess
      * @param message
      */
     public Result( boolean isSuccess, String message, T result ) {
            this.isSuccess = isSuccess;
            this.message = message;
            this.result = result;
     }     
     /**
      * 返回一个成功的result，包含要返回的对象
      * @param result 结果对象
      */
     public Result( T result ){
            this.isSuccess =true;
            this.setResult (result );
     }
     /**
      * 创建一个失败的result，不包含返回对象
      * @param msg
      * @return
      */
     public static Result <Boolean > fail( String msg ){
            return new Result <Boolean >(false, msg);
     }   
     /**
      * 创建一个成功的result，不包含返回对象
      * @param msg
      * @return
      */
     public static Result <Boolean > succeed( String msg ){
            return new Result <Boolean >(true, msg);
     }     
     /**
      * 创建一个成功的result，不包含返回对象
      * @return
      */
     public static Result <Boolean > succeed(){
            return new Result <Boolean >(true);
     }     
     public boolean isSuccess () {
            return isSuccess ;
     }
     public Result< T> setSuccess(boolean isSuccess) {
            this.isSuccess = isSuccess;
            return this ;
     }
     public boolean isFailed (){
            return !isSuccess;
     }
     public String getMessage() {
            return message ;
     }
     public Result< T> setMessage(String message) {
            this.message = message;
            return this ;
     }
     public T getResult() {
            return result ;
     }
     public Result< T> setResult(T result) {
            this.result = result;
            return this ;
     }
}
